% --- THE PROBLEM ---
\begin{frame}{The Problem: The Chaos of Indian Railways}
    \begin{alertblock}{For millions, planning a train journey is a nightmare.}
    \end{alertblock}
    \begin{itemize}
        \item \textbf{Information Overload:} Travelers juggle multiple apps and websites, wasting time and effort.
        \item \textbf{One-Size-Fits-All "Best" Routes:} Existing solutions ignore the nuanced priorities of travelers, optimizing only for speed or cost.
        \item \textbf{Last-Minute Travel is a Gamble:} The unreliable "Tatkal" system leaves travelers stranded or forced into expensive alternatives.
        \item \textbf{Safety is an Unquantified Risk:} There is no objective data to help travelers choose safer routes, a major concern for families and solo travelers.
    \end{itemize}
\end{frame}
